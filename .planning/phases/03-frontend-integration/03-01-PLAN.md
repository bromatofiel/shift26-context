1. Create `front/public/sw.js` with basic cache strategy:

```javascript
const CACHE_NAME = 'blindspot-v1'
const urlsToCache = [
  '/',
  '/search',
]

self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME)
      .then((cache) => cache.addAll(urlsToCache))
      .then(() => self.skipWaiting())
  )
})

self.addEventListener('activate', (event) => {
  event.waitUntil(
    caches.keys()
      .then((cacheNames) => {
        return Promise.all(
          cacheNames
            .filter((name) => name !== CACHE_NAME)
            .map((name) => caches.delete(name))
        )
      })
      .then(() => self.clients.claim())
  )
})

self.addEventListener('fetch', (event) => {
  // Network-first for API calls
  if (event.request.url.includes('/api/')) {
    event.respondWith(
      fetch(event.request)
        .catch(() => new Response(JSON.stringify({ error: 'Offline' }), {
          headers: { 'Content-Type': 'application/json' }
        }))
    )
    return
  }

  // Cache-first for static assets
  event.respondWith(
    caches.match(event.request)
      .then((response) => response || fetch(event.request))
  )
})
```

Create `front/app/share/page.tsx` to handle Android Share Target intents:

```typescript
import { redirect } from 'next/navigation'

interface SharePageProps {
  searchParams: Promise<{
    url?: string
    text?: string
    title?: string
  }>
}

export default async function SharePage({ searchParams }: SharePageProps) {
  const params = await searchParams

  // Android may send URL in 'text' field instead of 'url' field per W3C spec
  // Also check 'title' as some apps use different fields
  let sharedUrl = params.url || ''

  // If no direct URL, try to extract from text field
  if (!sharedUrl && params.text) {
    // params.text might contain "Check this out: https://..." or just the URL
    const urlMatch = params.text.match(/https?:\/\/[^\s]+/)
    sharedUrl = urlMatch ? urlMatch[0] : params.text
  }

  // Validate URL format
  if (!sharedUrl || !sharedUrl.startsWith('http')) {
    // Invalid URL - redirect to home
    redirect('/')
  }

  // Redirect to results page with URL for analysis
  redirect(`/results?url=${encodeURIComponent(sharedUrl)}`)
}
```

Key implementation details:
- Server Component (no 'use client') for immediate redirect
- Checks params.url first, then params.text (Android behavior varies by source app)
- Extracts URL from text field if embedded in sentence
- Validates URL starts with http/https
- Fallback to home page for invalid shares
- Redirects to /results?url={encoded_url} for analysis
